        val unlockedAt: Long
    )

    Create AchievementDao.kt:
    ```kotlin
    package hr.fipu.organizationtool.data

    import androidx.room.Dao
    import androidx.room.Insert
    import androidx.room.OnConflictStrategy
    import androidx.room.Query

    @Dao
    interface AchievementDao {
        @Insert(onConflict = OnConflictStrategy.IGNORE)
        suspend fun insert(achievement: AchievementEntity)

        @Query("SELECT * FROM achievements")
        suspend fun getAllUnlocked(): List<AchievementEntity>
    }
    ```

    OnConflictStrategy.IGNORE is correct: if the same achievement is checked again after already being unlocked, the insert is a no-op (idempotent).
